<script>
function toggleService() {
  const service = document.getElementById("serviceType").value;

  document.getElementById("cleaningSection").style.display =
    service === "cleaning" ? "block" : "none";

  document.getElementById("maintenanceSection").style.display =
    service === "maintenance" ? "block" : "none";
}

function calculatePrice() {
  let size = parseFloat(document.getElementById("size").value);
  let condition = parseFloat(document.getElementById("condition").value);
  let extras = parseFloat(document.getElementById("extras").value);

  let total = (size * condition) + extras;

  document.getElementById("total").innerText = total.toFixed(0);
}
</script>
